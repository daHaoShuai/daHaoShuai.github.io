#笛卡尔积查询：是两张表相乘的结果，如果左边表有M条记录，右边表有N条记录，则查询出M*N条记录，这样的结果中往往包含大量错误的数据，所以这种方法不经常使用
select * from dept02,emp02;
#内连接查询：查询左边表有且右边表也有的记录
select * from dept02,emp02 where dept02.id = emp02.dept02_id;
#左外连接查询：在内连接的基础上增加左边表有而右边表没有的记录
select * from dept02 left join emp02 on dept02.id = emp02.dept02_id;
#右外连接查询：在内连接的基础上增加右边表有而左边表没有的记录
select * from dept02 right join emp02 on dept02.id = emp02.dept02_id;
#全外连接查询：在内连接的查询基础上增加左边表有而右边表没有，右边表有而左边表没有的记录
select * from dept02 full join emp02 on dept02.id = emp02.dept_id;#mysql不支持的语句其他数据库支持
#MySQL这样全外连接查询
select * from dept02 left join emp02 on dept02.id = emp02.dept02_id union select * from dept02 right join emp02 on dept02.id = emp02.dept02_id;
#查询3号部门的名称和其中的员工姓名
select dept02.name 部门名称,emp02.name 员工名称 from dept02 inner join emp02 on dept02.id = emp02.dept02_id where dept02.id = 3;

#带IN关键字的子查询
#使用关键字进行子查询时，内层查询语句仅仅返回一个数据列，这个数据列中的值将供外层查询语句进行比较操作
#查询存在年龄为20岁的员工的部门
select dname from department where did in (select did from employee where age = 20);
#select语句中还可以使用NOT IN关键字，其作用跟IN相反
#查询不存在年龄为20岁的员工的部门
select dname from department where did not in (select did from employee where age = 20);
#exists关键字后面的参数可以是任意一个子查询，这个子查询的作用相当于测试，它不产生任何数据，只返回TRUE或者FALSE，返回结果为TRUE时，外层查询才会执行
select * from department where exists(select * from employee where age > 20);

#ANY关键字查询表示满足其中任意一个条件，它允许创建一个表达式对子查询的返回值进行比较，只要满足内层子查询中的任意一个比较条件就会返回一个结果
select * from department where did>any(select did from employee);

#带ALL关键字的子查询返回的结果需同时满足所有内层查询条件
#使用ALL关键字的子查询，查询满足条件的部门
select * from department where did>all(select did from employee);
#带比较运算符的子查询
#使用带比较运算符的子查询，查询李强是哪个部门的员工
select dname from department where did = (select did from employee where name="李强");

#创建一个存储过程判断变量V的值是否是null，输出对应信息
#IF语句
create procedure proc4()
BEGIN
     DECLARE v varchar(2);
     IF v IS NULL THEN select 'is null';
     ELSE select 'is not null';
     END IF;
END
call proc4;

drop procedure proc4;

create procedure proc4()
BEGIN
     DECLARE v varchar(2);
     set v = 'a';
     IF v IS NULL THEN select 'is null';
     ELSE select 'is not null';
     END IF;
END
call proc4;

#CASE语句
create procedure proc5()
begin
     declare v int;
     case v
          when 1 then select 'value is 1';
          when 2 then select 'value is 2';
          else select 'not 1 or 2';
     end case;
end
call proc5;

#loop语句
#LEAVE 跳出循环 、ITERATE 再次循环
create procedure doiterate()
begin
     declare p1 int default 0;
     my_loop:loop
                 set p1 = p1+1;
                 if p1<10 then iterate my_loop;
                 elseif p1>20 then leave my_loop;
                 end if;
                 select 'p1 is between 10 and 20';
     end loop my_loop;
end

repeat
      set id=id+1;
      until id>=10;
end repeat;

while i <10 do
      set i=i+1;
end while;

#修改存储过程
ALTER {PROCEDURE|FUNCTION} sp_name[characteristic...]
      CONTAINS SQL 表示子程序包含SQL语句，但不包含或写数据的语句；
      NO SQL 表示子程序中包含读数据的语句
      MODIFIES SQL DATA 表示子程序中包含写数据的语句
      SQL SECURITY{DEFINER|INVOKER} 指明谁有权限来执行
      DEFINER 表示只有定义者自己才能执行
      INVOKER 表示调用者可以执行
      COMMENT string 表示注释信息
#可以修改存储过程的特性，MySQL还不提供对已存在的存储过程代码修改，如果一定要修改，先删除后重新创建

#删除存储过程
DROP{PROCEDURE|FUNCTION}[IF EXISTS] sp_name
