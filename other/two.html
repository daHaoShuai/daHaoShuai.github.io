# scala学习

TIP

Scala 是一门多范式（multi-paradigm）的编程语言，设计初衷是要集成面向对象编程和函数式编程的各种特性。

Scala 运行在 Java 虚拟机上，并兼容现有的 Java 程序。

Scala 源代码被编译成 Java 字节码，所以它可以运行于 JVM 之上，并可以调用现有的 Java 类库。

scala编译

scalac *.scala
1

scala运行

scala *(文件名，不用加.class)
1

使用java运行

java -cp %SCALA_HOME%/lib/scala-library.jar; *

class User(name: String, age: Int) {//构造器直接这样写

      //用模板字符串输出
      println(s"${name}今年${age}岁了,在${User.school}上学")

}

object User {
  val school: String = "清华大学"//在伴生对象定义相当于java的静态参数

  def main(args: Array[String]): Unit = {
    val user = new User("小明", 20)
    user.printInfo()

}

# 条件循环

object Test04 {
  def main(args: Array[String]): Unit = {
    //    范围遍历 for(i <- 开始 to 结束)
    for (i <- 1 to 10) println(s"第${i}次循环")
    //    相当于for(i <- 1.to(10)) println(s"第${i}次循环")
    //  scala中方法的.在方法有参数时可以省略用空格代替,参数只有一个括号也能省略,如下
    val student = new Student()
    student eat "🐟"
    println("-----------------")
    //    for (i <- Range(1,10)) println(s"第${i}次循环")//相当于[1-10)左闭右开区间
    for (i <- 1 until 10) println(s"第${i}次循环") //相当于[1-10)左闭右开区间,跟Range(1,10)效果一样
    //    集合遍历
    println("-----------------")
    for (i <- Array(22,44,55,66,77)) println(i)//遍历数组
    for (i <- List(22,44,55,66,77)) println(i)//遍历集合
    for (i <- Set(22,44,55,66,77)) println(i)//遍历集合
  }
}

object Test01 {
  def main(args: Array[String]): Unit = {
    //    循环守卫
    //    for (i <- 1 to 10) if (i != 5) println(i)//普通写法,当i等于5时不打印

    //    倒序循环
    println("=================")

    for (i <- 1 to 10 reverse) println(i)//reverse写法,相当于上面
  }
}

# 循环返回值

object Test04 {
  def main(args: Array[String]): Unit = {
    //    循环返回值
    val result: Unit = for (i <- 1 to 10) println(i)
    println(s"result = $result") //result = () 默认返回值为空
    val res = for (i <- 1 to 10) yield i

    println(s"res1 => $res1") //res1 => Vector(1, 4, 9, 16, 25, 36, 49, 64, 81, 100)

# 中断循环（scala没有bleak关键字，不能直接中断循环）

import scala.util.control.Breaks

object Test06 {
  def main(args: Array[String]): Unit = {
    //    采用抛出异常的方式退出循环
    try {
      for (i <- 1 to 5) {
        if (i == 3) throw new RuntimeException()
        println(i)
      }
    } catch {
      case e: Exception => //什么都不做
    }
    println("================")
    //    使用Scala中的Breaks类的break方法,实现异常的抛出和捕获
    Breaks.breakable( //标记
      for (i <- 1 to 5) {
        if (i == 3) Breaks.break()
        println(i)
      }
    )
  }
}

# scala函数式编程

例子

object Test02 {
  def main(args: Array[String]): Unit = {
//    定义一个匿名函数
    val fun = (name: String) =>{println(name)}
    fun("哈哈哈")
//    定义一个函数,以函数作为参数传入
    def f(func: String => Unit): Unit = func("呜呜呜呜")
    f(fun)//传入fun函数
    f(name => println(name))//传入匿名函数
//    如果参数只出现一次,则参数可以省略并且后面的参数可以用_代替
    f(println(_))
//    如果可以推断出,当前传入的println是一个函数体,而不是调用语句,则可以省略_
    f(println)
  }
}

# 二元运算函数例子

object Test03 {
  def main(args: Array[String]): Unit = {
    //    定义一个"二元运算函数",只操作1和2这2个数,具体运算通过参数传入
    def oneAndTwo(fun: (Int, Int) => Int): Int = {
      fun(1, 2)
    }

    //    定义操作方法
    val add = (a: Int, b: Int) => a + b //加
    val minus = (a: Int, b: Int) => a - b //减
    println(oneAndTwo(add))
    println(oneAndTwo(minus))
    //    匿名函数简化
    println(oneAndTwo((a: Int, b: Int) => a + b))
    println(oneAndTwo((a: Int, b: Int) => a - b))
//    更加精简
    println(oneAndTwo(_ + _))
    println(oneAndTwo(_ - _))//a - b
    println(oneAndTwo(-_ + _))//-a + b => b -a
  }
}

# 二元计算函数例子优化

object Test04 {
  def main(args: Array[String]): Unit = {
    //    定义二元计算函数
    def dualEval(op: (Int, Int) => Int, a: Int, b: Int): Int = op(a, b)
//    匿名函数写法
    println(dualEval(_+_,1,2))//加法
    println(dualEval(_-_,1,2))//减法 a - b
    println(dualEval(-_+_,1,2))//减法 b - a
  }
}

# 数组操作

def main(args: Array[String]): Unit = {
  val arr: Array[Int] = Array(12, 34, 42, 15, 45)
  //    对数组进行处理,将操作抽象出来,处理完成之后的结果返回一个新的数组
  def arrayOp(array: Array[Int], op: Int => Int): Array[Int] = for (elem <- array) yield op(elem)
  //    定义一个加一操作(匿名函数方式)
  val newArr = arrayOp(arr,_+1)
  println(newArr.mkString(","))
  //    定义一个平方操作(匿名函数方式)
  val nnewArr = arrayOp(arr,i => i * i)
  println(nnewArr.mkString(","))
}

# 递归

import scala.annotation.tailrec

object Test01 {
  def main(args: Array[String]): Unit = {
    println(fact(100))
    println(tailFact(100))

    //    递归实现阶乘,普通写法
    def fact(n: Int): Int = {
      if (n == 0) return 1
      fact(n - 1) * n
    }

    //    递归实现阶乘,尾递归优化
    def tailFact(n: Int): Int = {
      @tailrec //检查尾递归,确保写出的尾递归是正确的
      def loop(n: Int, result: Int): Int = {
        if (n == 0) return result
        loop(n - 1, result * n)
      }
      loop(n, 1)
    }
  }
}
