public class MainActivity extends AppCompatActivity {
    TextView textView;
    int num = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        textView = findViewById(R.id.tv1);
    }

    //增加
    public void btnAdd(View view) {
        textView.setText(String.valueOf(++num));
    }

    //减少
    public void btnSub(View view) {
        String str = (String) textView.getText();
        int nums = Integer.parseInt(str);
        if (nums == 0) {
            Toast.makeText(this, "到底了....", Toast.LENGTH_SHORT).show();
        } else {
            textView.setText(String.valueOf(num--));

        }
    }
}

# EditText编辑框

TIP

android:hint="用户名/手机号"：默认提示词

android:textColorHint="#00ff00"：设置默认提示词颜色

android:minLines="3"：设置最小的行数

android:maxLines="3"：设置最大的行数，当输入的内容超过最大行数，文字会自动往上滚动。

设置编辑框的输入类型： android:inputType="date"：日期

android:inputType="time"：时间

android:inputType="textEmailAddress":邮箱

android:inputType="textPassword"：文本密码

android:inputType="numberPassword"：数字密码

设置编辑框的焦点： android:selectAllOnFocus="true"

# ImageView图片视图

TIP

在使用ImageView组件显示图像时，通常需要将显示的图像放置在res/drawable或者res/mipmap目录下即可。

android:scaleType=""：用于设置所显示图片如何缩放或移动以适应ImageView的大小，属性值可以是：

1.matrix：表示进行缩放

2.fitXY：图片的横向，纵向独立缩放，使得该图片完全适应于ImageView的大小。

3.fitStart：图片的纵横比缩放图片，直到该图片显示在ImageView中，缩放完成后图像在ImageView的左上角。

4.fitCenter：图片的纵横比缩放图片，直到该图片显示在ImageView中，缩放完成后图像在ImageView的中间。

5.fitEnd：图片的纵横比缩放图片，直到该图片显示在ImageView中，缩放完成后图像在ImageView的右下角。

6.center：以图片的中心点和ImageView的中心为基准，按照图片的原来大小居中显示，不缩放，用ImageView的大小截取图片的居中部分。

7.centerCrop：使得图片完全覆盖ImageView的大小。

8.centerInside：完全显示该图片，

android:adjustViewBounds="true"：自动调整图片的边界（宽和高）

true：自动调整边界

false：默认值。不需要自动调整边界

<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    xmlns:app="http://schemas.android.com/apk/res-auto"
    xmlns:tools="http://schemas.android.com/tools"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:orientation="vertical"
    tools:context=".MainActivity">

    <RelativeLayout
        android:layout_width="match_parent"
        android:layout_height="100dp"
        android:background="#ff0">

        <ImageView
            android:id="@+id/image1"
            android:layout_width="wrap_content"
            android:layout_height="wrap_content"
            android:adjustViewBounds="true"
            android:src="@drawable/img1" />

        <TextView
            android:layout_width="wrap_content"
            android:layout_height="wrap_content"
            android:layout_toRightOf="@id/image1"
            android:text="超人"
            android:textSize="32sp" />
    </RelativeLayout>

    <RelativeLayout
        android:layout_width="match_parent"
        android:layout_height="100dp"
        android:background="#f00">

        <ImageView
            android:id="@+id/image2"
            android:layout_width="wrap_content"
            android:layout_height="wrap_content"
            android:src="@mipmap/img1" />
            
        <TextView
            android:layout_width="wrap_content"
            android:layout_height="wrap_content"
            android:layout_toRightOf="@id/image2"
            android:text="超人"
            android:textSize="32sp" />
    </RelativeLayout>
</LinearLayout>

public class MainActivity extends AppCompatActivity {

    String[] data = new String[]{
            "江雪",
            "[作者]柳宗元",
            "千山鸟飞绝，",
            "万径人踪灭。",
            "孤舟蓑笠翁，",
            "独钓寒江雪。"
    };
    TextSwitcher textSwitcher;
    int index=0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        textSwitcher = findViewById(R.id.textSwitcher);
        textSwitcher.setFactory(new ViewSwitcher.ViewFactory() {
            @Override
            public View makeView() {
                //创建文本框显示诗句的内容
                TextView textView = new TextView(MainActivity.this);
                textView.setTextSize(30);//字体的大小
                textView.setTextColor(Color.BLUE);//字体的颜色
                FrameLayout.LayoutParams params = new FrameLayout.LayoutParams(ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT);
                params.gravity = Gravity.CENTER;
                textView.setLayoutParams(params);
                return textView;
            }
        });
    }

    @Override
    public boolean onTouchEvent(MotionEvent event) {
        //判断手机单击屏幕
        if(event.getAction()==MotionEvent.ACTION_DOWN){
            //切换显示的诗句
            textSwitcher.setText(data[index++%data.length]);
        }

public class MyAdapter extends BaseAdapter {
    Context context;
    String[] fruitName;
    int[] imageName;

    public MyAdapter(Context context, String[] fruitName, int[] imageName) {
        this.context = context;
        this.fruitName = fruitName;
        this.imageName = imageName;
    }

    //绑定列表的子项数目
    @Override
    public int getCount() {
        return fruitName.length;
    }

    //根据索引获取该位置的对象
    @Override
    public Object getItem(int position) {
        return null;
    }

    //获取子项的ID
    @Override
    public long getItemId(int position) {
        return 0;
    }

    //获取子项要显示的界面
    @Override
    public View getView(int position, View view, ViewGroup parent) {
        LayoutInflater inflater = LayoutInflater.from(context);
        View itemView;
        ViewHolder viewHolder;
        if (view == null) {
            itemView = inflater.inflate(R.layout.item, parent, false);
            viewHolder = new ViewHolder();
            viewHolder.imageView = itemView.findViewById(R.id.itemImage);
            viewHolder.textView = itemView.findViewById(R.id.itemName);
            itemView.setTag(viewHolder);
        } else {
            itemView = view;
            viewHolder = (ViewHolder) itemView.getTag();
        }
        viewHolder.imageView.setImageResource(imageName[position]);
        viewHolder.textView.setText(fruitName[position]);
        return itemView;
    }

    class ViewHolder {
        ImageView imageView;
        TextView textView;
    }

}

新建RecyclerView适配器

MyAdapter.java

public class MyAdapter extends RecyclerView.Adapter<MyAdapter.ViewHolder> {
    private List<Fruit> fruitList;

    public MyAdapter(List<Fruit> fruitList) {
        this.fruitList = fruitList;
    }

    @Override
    public MyAdapter.ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
        View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.item, parent, false);
        ViewHolder viewHolder = new ViewHolder(view);
        return viewHolder;
    }

    @Override
    public void onBindViewHolder(MyAdapter.ViewHolder holder, int position) {
        Fruit fruit = fruitList.get(position);
        holder.imageView.setImageResource(fruit.getImageId());
        holder.textView.setText(fruit.getFruitName());

    }

    @Override
    public int getItemCount() {
        return fruitList.size();
    }


    public class ViewHolder extends RecyclerView.ViewHolder {
        ImageView imageView;
        TextView textView;

        public ViewHolder(View itemView) {
            super(itemView);
            imageView = itemView.findViewById(R.id.itemImage);
            textView = itemView.findViewById(R.id.itemName);
        }
    }

}
