    # 选项菜单

    onCreateOptionMenu:选项菜单创建时执行(执行一次)

    onPrepareOptionsMenu:选项菜单打开时执行(每次打开都会执行)

    onOptionsItemSelected:选项菜单的菜单项被选择的时执行

    onOptionsMenuClosed:x选项菜单被关闭时执行(早期android适合)

    MainActivity.java

    # 上下文菜单

    onCreateContextMenu创建上下文菜单

    onContextItemSelected上下文菜单被选中

    onContextMenuClosed上下文菜单关闭

    MainActivity.java

    public class MainActivity extends AppCompatActivity {
    TextView textView;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        textView = findViewById(R.id.tv);
        registerForContextMenu(textView);//进行注册
    }

    @Override
    public void onCreateContextMenu(ContextMenu menu, View v, ContextMenu.ContextMenuInfo menuInfo) {
        super.onCreateContextMenu(menu, v, menuInfo);
        MenuInflater inflater = getMenuInflater();
        inflater.inflate(R.menu.main, menu);
    }

    @Override
    public boolean onContextItemSelected(MenuItem item) {
        switch (item.getItemId()) {
            case R.id.font_red:
                textView.setBackgroundColor(Color.RED);
                break;
            case R.id.font_blue:
                textView.setBackgroundColor(Color.BLUE);
                break;
            case R.id.font_green:
                textView.setBackgroundColor(Color.GREEN);
                break;
        }

        return super.onContextItemSelected(item);
    }

    @Override
    public void onContextMenuClosed(Menu menu) {
        super.onContextMenuClosed(menu);
    }
}

    public void btn4(View view) {
    //选项中的数据
    final String[] strings = {"唐僧", "悟空", "八戒", "沙僧", "白龙马"};
    //描述选项是否选中的数组
    final boolean[] checkItems = {false, true, false, false, false};
    new AlertDialog.Builder(MainActivity.this)
            .setTitle("遇到白骨精，谁去迎战?")
            .setMultiChoiceItems(strings, checkItems, new DialogInterface.OnMultiChoiceClickListener() {
                @Override
                public void onClick(DialogInterface dialog, int which, boolean isChecked) {

        @Override
        public void onClick(DialogInterface dialog, int which) {

            StringBuilder builders = new StringBuilder();

                if (checkItems[j]) {
                    builders.append(strings[j]);
                }
            }
            Toast.makeText(MainActivity.this, "打野的是:" + builders.toString(), Toast.LENGTH_SHORT).show();
        }
    }).create().show();
}

    # 自定义TextView

    MyTextView.java

    public class MyTextView extends AppCompatTextView {
    private int myWidth;
    private int myTranslateLength;
    private Paint myPaint;
    private Matrix myMatrix;
    private LinearGradient myGradient;

    public MyTextView(Context context, AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
    }

    public MyTextView(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    public MyTextView(Context context) {
        super(context);
    }

    @Override
    protected void onSizeChanged(int w, int h, int oldw, int oldh) {
        super.onSizeChanged(w, h, oldw, oldh);
        if (w != 0) {
            myWidth = w;
        } else {
            myWidth = getMeasuredWidth();
        }
        myPaint = getPaint();
        myGradient = new LinearGradient(-myWidth, 0, 0, 0, new int[]{Color.YELLOW,
                Color.GREEN, Color.MAGENTA, Color.GREEN, Color.RED},
                new float[]{0, 0.25f, 0.5f, 0.75f, 1}, Shader.TileMode.CLAMP);
        myPaint.setShader(myGradient);
        myPaint.setColor(Color.BLACK);
        myMatrix = new Matrix();
    }

    @Override
    protected void onDraw(Canvas canvas) {
        super.onDraw(canvas);
        if (myMatrix == null) return;
        myTranslateLength += myWidth / 15; //每次移动原来宽的1/15
        if (myTranslateLength > myWidth * 2) {
            myTranslateLength -= myWidth * 2;
        }
        myMatrix.setTranslate(myTranslateLength, 0);
        myGradient.setLocalMatrix(myMatrix);
        postInvalidateDelayed(100); //间隔0.1秒刷新
    }
}

    public class MainActivity extends AppCompatActivity {
    ListView listView;
    DrawerLayout drawerLayout;
    ImageView imageView;
    TextView textView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        ActionBar actionBar = getSupportActionBar();
        actionBar.hide();

        listView = findViewById(R.id.myListView);
        drawerLayout = findViewById(R.id.myDrawerLayout);
        imageView = findViewById(R.id.myImageView);
        textView = findViewById(R.id.myTextView);
        initData();

    }

    private void initData() {
        final List<String> myList = new ArrayList<>();
        myList.add("寻龙传说");
        myList.add("人潮汹涌");
        myList.add("阿凡达达");
        myList.add("哥斯拉啦");
        ArrayAdapter<String> adapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, myList);
        listView.setAdapter(adapter);
        listView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
                switch (position) {//切换电影海报
                    case 0:
                        imageView.setImageResource(R.drawable.image2);
                        break;
                    case 1:
                        imageView.setImageResource(R.drawable.image3);
                        break;
                    case 2:
                        imageView.setImageResource(R.drawable.image4);
                        break;
                    case 3:
                        imageView.setImageResource(R.drawable.image5);
                        break;
                }
                textView.setText(myList.get(position));//获取电影名称
                showDrawerLayout();
            }
        });
        //drawerLayout.openDrawer(Gravity.LEFT);//侧滑打开
    }

    //显示侧滑菜单
    public void showDrawerLayout() {
        if (!drawerLayout.isDrawerOpen(Gravity.LEFT)) {
            drawerLayout.openDrawer(Gravity.LEFT);
        } else {
            drawerLayout.closeDrawer(Gravity.LEFT);
        }
    }
}
